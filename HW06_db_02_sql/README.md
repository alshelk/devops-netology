Приведите получившуюся команду или docker-compose манифест.

docker-compose манифест:

```bash
$ cat docker-compose.yaml 
version: '2.5.1'

networks:
  monitoring:
    driver: bridge

volumes:
    pg_data: {}
    pg_backup: {}

services:
  pg_db:
    image: postgres:12
    restart: always
    environment:
      POSTGRES_PASSWORD: passw0rd
    volumes:
      - pg_data:/var/lib/postgresql/data
      - pg_backup:/backup
    ports:
```

```bash

```
